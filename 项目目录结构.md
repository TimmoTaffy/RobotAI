│   ├── conftest.py                          # pytest配置与fixtures

│   ├── test_trajectory_tracker.py           # 轨迹跟踪测试(7个函数)
│   ├── test_mpc_controller.py               # MPC控制测试(7个函数)
│   ├── test_high_level_fsm.py               # 基础状态机测试(10个函数)
│   ├── test_intelligent_fsm.py              # 智能状态机测试(11个函数)
│   ├── test_integration.py                  # 集成测试
│   ├── visualize_comprehensive_planning.py  # 综合规划可视化
│   ├── visualize_control_comparison.py      # 控制器对比可视化
│   └── visualize_tracking_comparison.py     # 跟踪器配置对比可视化

├── config-guide.md         # 配置指南
│
├── main.py                 # 主程序入口

│
├── run_tests.py            # 统一测试工具

├── notebooks/              #Jupyter演示
├── reports/                #测试报告
